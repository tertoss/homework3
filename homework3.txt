#2 задание

stages = []
num_stages = 9  # количество стадий

for i in range(1, num_stages + 1):
    stage = input(f"Введите стадию развития человека {i} (последняя стадия - Homo sapiens sapiens): ")
    stages.append(stage)

print(" => ".join(stages))
